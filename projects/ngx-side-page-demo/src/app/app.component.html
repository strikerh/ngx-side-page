Basic Usage


// Import the module
import {{ '{' }} SidePageService {{ '}' }} from 'ngx-side-page';

// Inject in your component
constructor(private sidePageService: SidePageService) {{ '{' }}{{ '}' }}

// Open a side page
openSidePage() {{ '{' }}
  const ref = this.sidePageService.openSidePage(
    'uniqueKey',
    YourComponent,
    {{ '{' }}
      width: '400px',
      position: 'end',
      main_data: {{ '{' }} yourData: 'value' {{ '}' }}
    {{ '}' }}
  );
{{ '}' }}

Configuration Options

Option Type Default Description
position string 'end' Position of side page: 'start' (left) or 'end' (right)
width string '100%' Width of the side page
maxWidth string '400px' Maximum width of the side page
minWidth string '250px' Minimum width of the side page
showCloseBtn boolean true Whether to show the close button
hasBackdrop boolean true Whether to show a backdrop
panelClass string undefined Custom CSS class for the side page panel
backdropClass string undefined Custom CSS class for the backdrop overlay
zIndex number 1000 Z-index of the side page
main_data any - Data to pass to the component
